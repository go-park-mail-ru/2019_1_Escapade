# FAQ
# Для указания переменной используйте шаблон 
# {название переменной} = {значение переменной}
#   - знак равенства должен быть отделен пробелами
#   - значение переменной: 
#     - для строк двойные кавычки не нужны. 
#   - название переменной:
#     - должно содержать ключевое слово (для поиска
#       grep, например addr)
# Для комментирования используется '#'

# Адреса серверов, первый будет менеджером, 
# остальные рабочими
# (grep addr)
addr1 = 209.97.156.145
addr2 = 64.225.23.106
addr3 = 167.172.129.75

# Путь до ssh ключа(по умолчанию ~/.ssh/) 
# (grep path)
path = .ssh/id_rsa

# названия машин 
# (grep name)
name1 = api1
name2 = api2
name3 = api3
